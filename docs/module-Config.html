(inner) initializeService(config)

Description:
  • Initializes the service with the given configuration. This function must be called before using any other functions in this library.
Source:
  • config.js, line 43
Example
// Initialize the service in your app.js
initializeService({
  sanityConfig: {
    token: 'your-sanity-api-token',
    projectId: 'your-sanity-project-id',
    dataset: 'your-dataset-name',
    version: '2021-06-07',
    debug: true,
    useCachedAPI: false
  },
  railcontentConfig: {
    token: 'your-user-api-token',
    userId: 'current-user-id'
  }
});
